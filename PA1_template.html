aveges <- steps_by_interval$avg_steps
names(aveges) <- steps_by_interval$interval

Are there differences in activity patterns between weekdays and weekends?

Mutate data

actitvity_by_daytype <-
    new_steps %>%
        mutate(
            daytype = as.factor(
                ifelse(
                    weekdays(
                        as.Date(date),
                        abbreviate = TRUE
                    ) %in% c("Sat", "Sun"),
                    "weekend",
                    "weekday"
                )
            )
        )
